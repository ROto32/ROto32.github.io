      Lua Example Code for Key Validation

      This example demonstrates how to fetch and validate a key using Lua.

      
// Lua Example Code
local JSONBIN_API_KEY = "$2a$10$nINI7pBO7a.W7H0knci4VehSa2YblgI/5pz/FMsoroByYUqKO./Ju" -- Your JSONBin API Key
local BIN_ID = "672e5527e41b4d34e450f833" -- Your JSONBin Bin ID

local function fetchKeys()
    local url = "https://api.jsonbin.io/v3/b/" .. BIN_ID .. "/latest"
    local headers = {
        ["X-Master-Key"] = JSONBIN_API_KEY,
        ["Content-Type"] = "application/json"
    }

    local success, response = pcall(function()
        return request({
            Url = url,
            Method = "GET",
            Headers = headers
        })
    end)

    if not success then
        print("Request failed with error:", response)
        return nil
    end

    if not response then
        print("Response is nil, possibly a network or API error.")
        return nil
    end

    if response.StatusCode ~= 200 then
        print("Request did not succeed. Status:", response.StatusMessage or "Unknown", "Code:", response.StatusCode)
        return nil
    end

    local data = game:GetService("HttpService"):JSONDecode(response.Body)
    if not data or not data.record or not data.record.keys then
        print("Invalid data format in response:", response.Body)
        return nil
    end

    print("Data retrieved successfully:", data) -- Debugging line to check data structure
    return data.record.keys
end

local function validateKey(userKey)
    local keys = fetchKeys()
    if not keys then
        print("Could not fetch keys.")
        return false
    end

    for _, keyData in ipairs(keys) do
        if keyData.key and keyData.expiresAt then
            if keyData.key == userKey and keyData.expiresAt > os.date("!%Y-%m-%dT%H:%M:%S") then
                return true -- Key is valid
            end
        else
            print("Key data is missing required fields.")
            return false
        end
    end

    return false -- Key is invalid
end

local key = "EXAMPLE_KEY"
if validateKey(key) then
    print("Key is valid!")
else
    print("Invalid key.")
end
